        % 使用 postaction 来在线条绘制完成后再执行一个动作

            decorate,
            decoration={
                markings,
                mark=at position 0.5 with {\arrow{>}}, % 在中点放置一个箭头
            },
        },
        draw=none, % 不绘制线本身，只绘制箭头
    },
    % 动量标签位置调整

        }
    }
}
